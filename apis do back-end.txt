apis  do back-end:


- api login 
	window.api.login.login(email, senha); 
	retorna um objeto com usuário(id, nome, email, tipo)
(ao fazer login use a api setCurrentUser para settar  o usuário atual)
	objeto "usuário"
	window.api.user.setCurrentUser(usuario);
		retorna true ou false
	window.api.user.getCurrentUser(); 
		retorna o objeto usuario


- api cadastro de usuário
	window.api.funcionario.cadastrarFuncionario(nome, cpf, email, cargo, senha)
	retorna um objeto com usuário(id, nome, email, tipo, senha: true)
- api get user citada no login


- api cadastro de categoria 
	window.api.categoria.cadastrarCategoria(nome, status)
	retorna true ou false

- api get categorias
	window.api.categoria.getCategorias()
	retorna um objeto com id, nome e status
	ordenado pelo nome


- api cadastro de produtos
		objeto = {
			nome,
			preco,
			idCategoria
			descrição
			}
	window.api.produto.cadastrarProduto(objeto);
	retorna true ou false



- api  get produtos pelo id categoria

	window.api.produto.getProdutosPorCategoria(idCategoria); 
		retorna um objeto com os  produtos(id, nome, preco, descrição)

- api atualizar status produto	
	
		window.api.produto.mudarStatus(idProduto);
			retorna o produto atualizado

- api get todos os produtos
	window.api.produto.getTodosProdutos();
	retorna um objeto com todos os produtos(id, nome. preco, descrição, categoria_id)


- api get funcionários
	window.api.funcionario.getFuncionario(tipoFuncionario);
	"Gerente" ou "Garçom"


- api cadastrar mesas
	objeto = {	numero: "",
			status: "",
			cadeiras: "" 
			}
	window.api.mesas.cadastrarMesas(objeto);
		retorna true


- api get mesas
	window.api.mesas.listarMesas(); 
		return um array de objetos, status, numero, n_cadeiras

- api remover mesa 
		window.api.mesa.remover(mesa_numero);
		return true

		
- api verificar pedidos de uma mesa
	window.api.mesas.verificarMesaPedido(numeroMesa);
	return objeto pedido id, mesa_numero, data criação, status, valor total, id_funcionario


- api criar pedido 
		(verificação de mesa já existente na query)
	window.api.pedido.registrarPedido(numeroMesa, idGarcom)
		retorna true 



- api adicionar produtos a um pedido
			
		window.api.pedido.adicionarProdutosPedido(idPedido, idProduto, quantidade); 

				retorna objeto item =  {
					pedido_id,
        			produto_id,
        			quantidade,
				}

- api listar itens do pedido 
		window.api.pedido.listarItensPedido(id_pedido);
			retorna um array de objetos {
						id, nome, preco, descrição
							}
							(qualquer coisa da um console.log pra ver o tipo de retorno)



- api editar pedido 

		window.api.pedido.editarPedido(idpedido, dadosAtualizados);

	dadosAtualizados = {	
				 	mesa_numero,
     				status,
      				id_funcionario,
				}

		return true ou false


- api excluir item do pedido 
		window.api.pedido.removerItem(idPedido, idProduto, quantidade)

		return true ou false



- api fechar pedido 
		window.api.pedido.fecharPedido(idpedido, valor_total);
		
		return objeto pedido


- api fazer pagamento 
		window.api.pagamento.cadastrarPagamento(idPedido, tipoPagamento, dividido, valorPago);
		idpedido = id do pedido fechado 
		tipoPagamento = "dinheiro', "debito", "credito" ou "pix"
		dividido: true ou false, (se o pagamento foi dividido)
		valorPago: valor pago 

			retorna o objeto pagamento(id, pedido_id, tipo_pagamento, dividido, valorPago,)



- api todos os pagamento (pra um historio ou algo assim)
		window.api.pagamento.todosOsPagamentos();
			retorna um array com cada objeto (id, pedido_id, tipo_pagamento, dividido, valor_pago )
		


- api buscarPagamentoPorId do pedido

		window.api.pagamento.buscarPagamentoPorPedidoId(id_pedido);
			return objeto pedido(id, pedido_id, tipo_pagamento, dividido, valor_pago)
